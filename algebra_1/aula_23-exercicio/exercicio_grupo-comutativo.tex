                $x^2 = e$, para todo $x\in G$, ent{\~a}o $G$ {\'e} um grupo abeliano.

                \item Mostre que se $x\in G$ {\'e} tal que $x^2 = x$, ent{\~a}o $x$ {\'e} o elemento neutro.
            \end{enumerate}
        \end{exercicio}
