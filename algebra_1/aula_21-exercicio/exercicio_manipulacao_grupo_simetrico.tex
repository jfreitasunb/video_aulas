            Considere o grupo $S_4$.
            \begin{enumerate}[label=({\alph*})]
                \item Determine elementos $f$, $g \in S_4$ tais que
                    \[
                        (f \circ g)^4 \ne f^4 \circ g^4.
                    \]
                \item Para o elemento
                    \[
                        \sigma = \begin{pmatrix}
                                    1 & 2 & 3 & 4\\
                                    1 & 3 & 2 & 4
                                 \end{pmatrix}
                    \]
                encontre o menor $l \ge 0$ tal que $\sigma^l = 1$, onde
                \[
                    1 = \begin{pmatrix}
                        1 & 2 & 3 & 4\\
                        1 & 2 & 3 & 4
                    \end{pmatrix}.
                \]
            \end{enumerate}
        \end{exercicio}
